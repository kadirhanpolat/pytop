- Baseline reports committed to `docs/PERFORMANCE.md`

**Tech Stack:** 
- `cProfile` (stdlib, deterministic profiling)
- `tracemalloc` (stdlib, memory tracking)
- `py-spy` (flamegraph visualization, external but optional)
- `pstats` (stdlib, call statistics)
- `pytest` (fixtures for benchmark collection)

# src/pytop/_internal/profiling.py
import cProfile
import pstats
import tracemalloc
from functools import wraps
from typing import Any, Callable, Optional, TypeVar, Tuple
from dataclasses import dataclass
from io import StringIO
import time

T = TypeVar("T")

@dataclass
class ProfileStats:
    """Container for profiling results."""
    function_name: str
    total_time: float
    call_count: int
    caller_info: Optional[str]
    peak_memory_mb: Optional[float]
    top_5_callers: list[Tuple[str, int, float]]  # (func_name, ncalls, cumtime)
    raw_profile_data: Optional[dict]  # serializable pstats summary

def profile_call(
    track_memory: bool = False,
    track_caller: bool = False,
    enable_by_default: bool = True
) -> Callable[[Callable[..., T]], Callable[..., Tuple[T, ProfileStats]]]:
    """
    Decorator that profiles a function call via cProfile and optionally tracemalloc.
    
    Usage:
        @profile_call(track_memory=True)
        def my_function(x):
            return x * 2
        
        result, stats = my_function(5)
        print(stats.total_time, stats.peak_memory_mb)
    
    Args:
        track_memory: If True, track peak memory via tracemalloc
        track_caller: If True, extract caller info from stack
        enable_by_default: If False, skips profiling (for conditional profiling)
    
    Returns:
        Decorated function that returns (result, ProfileStats)
    """
    def decorator(func: Callable[..., T]) -> Callable[..., Tuple[T, ProfileStats]]:
        @wraps(func)
        def wrapper(*args: Any, **kwargs: Any) -> Tuple[T, ProfileStats]:
            if not enable_by_default:
                result = func(*args, **kwargs)
                return result, ProfileStats(
                    function_name=func.__name__,
                    total_time=0.0,
                    call_count=0,
                    caller_info=None,
                    peak_memory_mb=None,
                    top_5_callers=[],
                    raw_profile_data=None,
                )
            
            # Set up profiler and memory tracker
            profiler = cProfile.Profile()
            mem_snapshot_start = None
            mem_snapshot_end = None
            
            if track_memory:
                tracemalloc.start()
            
            start_time = time.perf_counter()
            profiler.enable()
            
            try:
                result = func(*args, **kwargs)
            finally:
                profiler.disable()
                end_time = time.perf_counter()
                
                if track_memory:
                    mem_snapshot_end = tracemalloc.take_snapshot()
                    tracemalloc.stop()
            
            # Extract stats
            stats_str = StringIO()
            ps = pstats.Stats(profiler, stream=stats_str)
            ps.sort_stats("cumulative")
            
            # Get top 5 callers
            top_5 = []
            for func_key in list(ps.stats.keys())[:5]:
                cc, nc, tt, ct, callers = ps.stats[func_key]
                func_name = f"{func_key[0]}:{func_key[1]}"
                top_5.append((func_name, nc, ct))
            
            # Memory tracking
            peak_memory_mb = None
            if track_memory and mem_snapshot_end:
                top_stat = mem_snapshot_end.statistics("lineno")[0]
                peak_memory_mb = top_stat.size / (1024 * 1024)
            
            # Build ProfileStats
            total_time = end_time - start_time
            call_count = ps.total_calls
            
            return result, ProfileStats(
                function_name=func.__name__,
                total_time=total_time,
                call_count=call_count,
                caller_info=None,  # TODO: extract from stack if track_caller=True
                peak_memory_mb=peak_memory_mb,
                top_5_callers=top_5,
                raw_profile_data={
                    "total_time": total_time,
                    "call_count": call_count,
                    "primitive_calls": ps.primitive_calls,
                }
            )
        
        return wrapper
    return decorator


def context_profile(
    func_name: str = "profiled_block",
    track_memory: bool = False
) -> Any:
    """
    Context manager for profiling a block of code.
    
    Usage:
        with context_profile(track_memory=True) as prof:
            # code to profile
        print(prof.stats.total_time)
    """
    class ProfileContext:
        def __init__(self):
            self.profiler = cProfile.Profile()
            self.stats = None
            self.mem_snapshot = None
            self.track_memory = track_memory
            self.start_time = None
        
        def __enter__(self):
            self.start_time = time.perf_counter()
            self.profiler.enable()
            if self.track_memory:
                tracemalloc.start()
            return self
        
        def __exit__(self, exc_type, exc_val, exc_tb):
            self.profiler.disable()
            end_time = time.perf_counter()
            
            if self.track_memory:
                self.mem_snapshot = tracemalloc.take_snapshot()
                tracemalloc.stop()
            
            # Collect stats
            stats_stream = StringIO()
            ps = pstats.Stats(self.profiler, stream=stats_stream)
            
            peak_memory_mb = None
            if self.mem_snapshot:
                top_stat = self.mem_snapshot.statistics("lineno")[0]
                peak_memory_mb = top_stat.size / (1024 * 1024)
            
            self.stats = ProfileStats(
                function_name=func_name,
                total_time=end_time - self.start_time,
                call_count=ps.total_calls,
                caller_info=None,
                peak_memory_mb=peak_memory_mb,
                top_5_callers=[],
                raw_profile_data=None,
            )
            
            return False
    
    return ProfileContext()
```

# src/pytop/_internal/profiling_report.py
import json
from dataclasses import dataclass, asdict
from typing import List, Optional
from datetime import datetime
from pytop._internal.profiling import ProfileStats

@dataclass
class ProfileReport:
    """Aggregated profiling report from multiple ProfileStats."""
    name: str
    stats_list: List[ProfileStats]
    timestamp: Optional[str] = None
    python_version: Optional[str] = None
    
    def __post_init__(self):
        if self.timestamp is None:
            self.timestamp = datetime.now().isoformat()
    
    @property
    def total_runtime(self) -> float:
        """Sum of all function times."""
        return sum(s.total_time for s in self.stats_list)
    
    @property
    def peak_memory_mb(self) -> float:
        """Maximum peak memory across all profiles."""
        peak = max(
            (s.peak_memory_mb for s in self.stats_list if s.peak_memory_mb),
            default=0.0
        )
        return peak
    
    def bottleneck_functions(self, top_n: int = 5) -> List[tuple]:
        """Top N slowest functions by cumulative time."""
        sorted_stats = sorted(self.stats_list, key=lambda s: s.total_time, reverse=True)
        return [
            (s.function_name, s.total_time, s.peak_memory_mb)
            for s in sorted_stats[:top_n]
        ]


def generate_markdown_report(report: ProfileReport) -> str:
    """Generate markdown-formatted profiling report."""
    lines = [
        f"# Profile Report: {report.name}",
        "",
        f"**Generated:** {report.timestamp}",
        f"**Total Runtime:** {report.total_runtime:.3f}s",
        f"**Peak Memory:** {report.peak_memory_mb:.1f} MB",
        "",
        "## Function Breakdown",
        "",
        "| Function | Time (s) | Calls | Peak Memory (MB) |",
        "|----------|----------|-------|------------------|",
    ]
    
    for stat in report.stats_list:
        mem_str = f"{stat.peak_memory_mb:.1f}" if stat.peak_memory_mb else "N/A"
        lines.append(
            f"| {stat.function_name} | {stat.total_time:.3f} | {stat.call_count} | {mem_str} |"
        )
    
    lines.extend(["", "## Top 5 Bottlenecks", ""])
    
    for i, (fname, ftime, fmem) in enumerate(report.bottleneck_functions(), 1):
        mem_str = f"{fmem:.1f} MB" if fmem else "N/A"
        lines.append(f"{i}. **{fname}**: {ftime:.3f}s ({mem_str})")
    
    return "\n".join(lines)


def generate_json_report(report: ProfileReport) -> str:
    """Generate JSON-formatted profiling report."""
    report_dict = {
        "name": report.name,
        "timestamp": report.timestamp,
        "total_runtime": report.total_runtime,
        "peak_memory_mb": report.peak_memory_mb,
        "function_count": len(report.stats_list),
        "stats": [
            {
                "function_name": s.function_name,
                "total_time": s.total_time,
                "call_count": s.call_count,
                "peak_memory_mb": s.peak_memory_mb,
            }
            for s in report.stats_list
        ],
    }
    return json.dumps(report_dict, indent=2)
```

# src/pytop/_internal/profiling_fixtures.py
from typing import Callable, List
from pytop._internal.profiling import profile_call, ProfileStats

class ProfileCollector:
    """Pytest fixture that collects ProfileStats across tests."""
    
    def __init__(self):
        self._stats: List[ProfileStats] = []
    
    def record(self, stats: ProfileStats):
        """Record a ProfileStats entry."""
        self._stats.append(stats)
    
    def collected_stats(self) -> List[ProfileStats]:
        """Return all collected stats."""
        return self._stats.copy()
    
    def clear(self):
        """Clear collected stats."""
        self._stats.clear()


def create_profile_benchmark(collector: ProfileCollector) -> Callable:
    """Factory for @profile_benchmark decorator that integrates with collector."""
    
    def profile_benchmark(track_memory: bool = False):
        def decorator(func: Callable):
            profiled_func = profile_call(track_memory=track_memory)(func)
            
            def wrapper(*args, **kwargs):
                result, stats = profiled_func(*args, **kwargs)
                collector.record(stats)
                return result, stats
            
            return wrapper
        return decorator
    
    return profile_benchmark

# src/pytop/_internal/benchmark_runner.py
"""
CLI tool to run profiling benchmarks and generate performance reports.

Usage:
    python -m pytop._internal.benchmark_runner --profile all --output docs/PERFORMANCE.md
    python -m pytop._internal.benchmark_runner --profile homology --output tmp.md
"""
import argparse
import subprocess
import json
from pathlib import Path
from pytop._internal.profiling_report import ProfileReport, generate_markdown_report

def run_benchmarks(profile_type: str = "all") -> list:
    """Run pytest benchmarks and capture profile stats."""
    
    marker = "benchmark" if profile_type == "all" else f"benchmark and {profile_type}"
    
    cmd = [
        "python", "-m", "pytest",
        "tests/profiling/",
        "-v",
        f"-m", marker,
        "--tb=short",
    ]
    
    result = subprocess.run(cmd, capture_output=True, text=True)
    
    if result.returncode != 0:
        print(f"Benchmark run failed:\n{result.stderr}")
        raise RuntimeError("Benchmark execution failed")
    
    return result.stdout


def generate_baseline_report(output_path: str):
    """Generate baseline performance report by running benchmarks."""
    
    print("Running profiling benchmarks...")
    benchmark_output = run_benchmarks("all")
    print(benchmark_output)
    
    # Create sample report (would aggregate from collected stats in real run)
    sample_report = ProfileReport(
        name="pytop baseline performance (Phase 17 P17.1)",
        stats_list=[],
    )
    
    md_text = generate_markdown_report(sample_report)
    
    output_file = Path(output_path)
    output_file.parent.mkdir(parents=True, exist_ok=True)
    output_file.write_text(md_text)
    
    print(f"✅ Baseline report written to {output_path}")


def main():
    parser = argparse.ArgumentParser(
        description="Run pytop profiling benchmarks and generate performance reports"
    )
    parser.add_argument(
        "--profile",
        choices=["all", "homology", "persistence", "khovanov"],
        default="all",
        help="Which benchmarks to run"
    )
